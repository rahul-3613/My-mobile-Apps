  <script>
    // Go to Page 2
    document.getElementById("enterBtn").onclick = function() {
      document.getElementById("page1").style.display = "none";
      document.getElementById("page2").style.display = "block";
    };

    // Exchange rates (BDT to others, demo values)
    const rates = {
      USD: 0.0091, EUR: 0.0085, INR: 0.76, GBP: 0.0072, JPY: 1.34,
      AUD: 0.014, CAD: 0.012, SGD: 0.012, CNY: 0.066, MYR: 0.043,
      THB: 0.34, PKR: 2.56, SAR: 0.034, AED: 0.033, KWD: 0.0028,
      BHD: 0.0032, NPR: 1.20, LKR: 2.72, ZAR: 0.17, TRY: 0.31
    };

    let historyCount = 1;

    function convert() {
      let amount = document.getElementById("amount").value;
      let currency = document.getElementById("currency").value;
      if (amount === "" || amount <= 0) {
        alert("Please enter a valid amount!");
        return;
      }
      let rate = rates[currency];
      let converted = (amount * rate).toFixed(2);
      document.getElementById("result").innerText = 
        amount + " BDT = " + converted + " " + currency;

      // Save to history
      let li = document.createElement("li");
      li.innerText = historyCount + ". " + amount + " BDT â†’ " + converted + " " + currency;
      document.getElementById("historyList").appendChild(li);
      historyCount++;
    }
  </script>
